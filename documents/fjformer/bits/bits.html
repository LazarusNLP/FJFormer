 15        a: jnp.ndarray,
 16        b: jnp.ndarray,
 17        *,
 18        precision=None,
 19        dot_general=lax.dot_general
 20) -> jnp.ndarray:
 21    """Quantized jax.numpy.matmul.
 22
 23    Args:
 24      a: Left-hand side of the matmul.
 25      b: Right-hand side of the matmul.
 26      precision: Indicates precision of a and b.
 27      dot_general: lax.dot_general by default. To use quantized matmul, the
 28        wrapper of q_dot_general in which TQs and `train` flag are provided
 29        should be passed into this function.
 30
 31    Returns:
 32      An array containing the result with the same dtype as 'a' and 'b'.
 33    """
 34    arraylike = (jax.Array, np.ndarray)
 35    if not isinstance(a, arraylike) or not isinstance(b, arraylike):
 36        raise TypeError(f"matmul requires array-like arguments, got {a} and {b}")
 37    for i, x in enumerate((a, b)):
 38        if ndim(x) < 1:
 39            msg = (f"matmul input operand {i} must have ndim at least 1, "
 40                   f"but it has ndim {ndim(x)}")
 41            raise ValueError(msg)
 42
 43    dtype = jnp.result_type(a.dtype, b.dtype)
 44    a = a.astype(dtype)
 45    b = b.astype(dtype)
 46
 47    a_is_mat, b_is_mat = (ndim(a) > 1), (ndim(b) > 1)
 48    a_batch_dims = shape(a)[:-2] if a_is_mat else ()
 49    b_batch_dims = shape(b)[:-2] if b_is_mat else ()
 50    num_batch_dims = _max(len(a_batch_dims), len(b_batch_dims))
 51    a_batch_dims = (None,) * (num_batch_dims - len(a_batch_dims)) + a_batch_dims
 52    b_batch_dims = (None,) * (num_batch_dims - len(b_batch_dims)) + b_batch_dims
 53
 54    # Dimensions to squeeze from the inputs.
 55    a_squeeze = []
 56    b_squeeze = []
 57
 58    # Positions of batch dimensions in squeezed inputs.
 59    a_batch = []
 60    b_batch = []
 61
 62    # Desired index in final output of each kind of dimension, in the order that
 63    # aqt_dot_general will emit them.
 64    idx_batch = []
 65    idx_a_other = []  # other = non-batch, non-contracting.
 66    idx_b_other = []
 67    for i, (ba, bb) in enumerate(zip(a_batch_dims, b_batch_dims)):
 68        if ba is None:
 69            idx_b_other.append(i)
 70        elif bb is None:
 71            idx_a_other.append(i)
 72        elif core.symbolic_equal_dim(ba, 1):
 73            idx_b_other.append(i)
 74            a_squeeze.append(len(idx_batch) + len(idx_a_other) + len(a_squeeze))
 75        elif core.symbolic_equal_dim(bb, 1):
 76            idx_a_other.append(i)
 77            b_squeeze.append(len(idx_batch) + len(idx_b_other) + len(b_squeeze))
 78        elif core.symbolic_equal_dim(ba, bb):
 79            a_batch.append(len(idx_batch) + len(idx_a_other))
 80            b_batch.append(len(idx_batch) + len(idx_b_other))
 81            idx_batch.append(i)
 82        else:
 83            raise ValueError("Incompatible shapes for matmul arguments: {} and {}"
 84                             .format(shape(a), shape(b)))
 85
 86    if a_is_mat:
 87        idx_a_other.append(num_batch_dims)
 88    if b_is_mat:
 89        idx_b_other.append(num_batch_dims + a_is_mat)
 90    perm = np.argsort(np.concatenate([idx_batch, idx_a_other, idx_b_other]))
 91
 92    a = lax.squeeze(a, tuple(a_squeeze))
 93    b = lax.squeeze(b, tuple(b_squeeze))
 94    out = dot_general(
 95        a,
 96        b, (((ndim(a) - 1,), (ndim(b) - 1 - b_is_mat,)), (a_batch, b_batch)),
 97        precision=precision)
 98    return lax.transpose(out, perm)

101def matmul_true_int8(lhs, rhs):
102    assert lhs.dtype == jnp.int8
103    assert rhs.dtype == jnp.int8
104    result = jnp.matmul(lhs, rhs, preferred_element_type=jnp.int32)
105    assert result.dtype == jnp.int32
106    return result

109def quant_int8(x):
110    return jnp.clip(jnp.round(x), -127, 127).astype(jnp.int8)

113def q_matmul_int8(a, w):
114
115    # Calibration. Calibration function is also customizable and injectable.
116    a_s = 127 / jnp.max(jnp.abs(a), axis=1, keepdims=True)
117    w_s = 127 / jnp.max(jnp.abs(w), axis=0, keepdims=True)
118
119    # int8 matmul with int32 accumulator
120    result = matmul_true_int8(quant_int8(a * a_s), quant_int8(w * w_s)) / (a_s * w_s)
121
122    return result
